数据结构在树上的应用

一些数据结构在树上应用的场景和技巧

树上单点修改和树链求和

hdu6200

题意：给定一张无向连通图，要求在支持连边的情况下，回答某条简单路径上的必经边条数。

思路：首先我们可以从一开始的图中任取一颗生成树，把多余的边存下来。一棵树的某条简单路径上的必经边条数就是这条路径的长度。
考虑每次增加一条边，必然会在原图上产生环，此时要把这个环所经过的树边的边权赋为零。
首先我们可以把边权转化为点权，然后用树状数组差分维护dfs序来实现求树上一点到根的权值和以及树上单点修改。
但每次修改一个环必然会超时，考虑到条边只会被赋值一次。我们可以利用并查集合并新增环上的点，即缩点保证每个点只修改一次。

这样总的时间复杂度就是$O(n \times log(n)+q \times log(n))$

代码：

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn = 1e5+10;
int T,n,m,q,head[maxn],nex[maxn<<1],to[maxn<<1],tot,tim;
struct DSU{
    int pre[maxn];
    void init(){ for(int i = 1; i <= n; i++) pre[i] = i; }
    int find(int u) { return u==pre[u]?u:pre[u]=find(pre[u]); }
    bool merge(int u,int v){
        u = find(u),v = find(v);
        if(u==v) return false;
        return pre[u] = v;
    }
}dsu[2];
struct BIT{
    int f[maxn];
    void init() { for(int i = 1; i <= n; i++) f[i] = 0; }
    void add(int p,int v) { for(; p <= n; p += p&(-p)) f[p] += v; }
    int query(int p) {
        int ret = 0;
        for(; p; p -= p&(-p))
            ret += f[p];
        return ret;
    }
}bit;
void addedge(int u,int v){
    to[++tot] = v,nex[tot] = head[u],head[u] = tot;
    to[++tot] = u,nex[tot] = head[v],head[v] = tot;
}
int fa[maxn],son[maxn],siz[maxn],top[maxn],dep[maxn],in[maxn],out[maxn];
void dfs(int u,int f){
    dep[u] = dep[f]+1,in[u] = ++tim;
    son[u] = 0,siz[u] = 1,fa[u] = f;
    for(int i = head[u]; i; i = nex[i]){
        if(to[i]==f) continue;
        dfs(to[i],u); siz[u] += siz[to[i]];
        if(siz[to[i]]>siz[son[u]]) son[u] = to[i];
    }
    out[u] = tim+1;
}
void dfs1(int u,int f){
    top[u] = u==son[f]?top[f]:u;
    for(int i = head[u]; i; i = nex[i])
        if(to[i]^f) dfs1(to[i],u);
}
int lca(int u,int v){
    while(top[u]^top[v])
        dep[top[u]]>dep[top[v]]?u=fa[top[u]]:v=fa[top[v]];
    return dep[u]>dep[v]?v:u;
}
void modify(int u,int v){
    if(dsu[1].find(u)==dsu[1].find(v)) return void();
    int p = lca(u,v);
    while(dep[fa[u]]>=dep[p]){
        if(dsu[1].find(u)^dsu[1].find(fa[u])){
            dsu[1].merge(u,fa[u]);
            bit.add(in[u],-1),bit.add(out[u],1),u=fa[u];
        }
        else u = dsu[1].find(u);
    }
    while(dep[fa[v]]>=dep[p]){
        if(dsu[1].find(v)^dsu[1].find(fa[v])){
            dsu[1].merge(v,fa[v]);
            bit.add(in[v],-1),bit.add(out[v],1),v=fa[v];
        }
        else v = dsu[1].find(v);
    }
}
int query(int u,int v){
    return bit.query(in[u])+bit.query(in[v])-2*bit.query(in[lca(u,v)]);
}
vector<pair<int,int> > vec;
void init(){
    tot = tim = 0,vec.clear();
    dsu[0].init(),dsu[1].init(),bit.init();
    for(int i = 1; i <= n; i++) head[i] = 0;
}
int kase;
int main(){
    for(scanf("%d",&T),kase=T;T;T--){
        scanf("%d%d",&n,&m); init();
        for(int i = 1; i <= m; i++){
            int u,v;
            scanf("%d%d",&u,&v);
            if(dsu[0].merge(u,v)) addedge(u,v);
            else vec.push_back({u,v});
        }
        dfs(1,0),dfs1(1,1);
        for(int i = 2; i <= n; i++)
            bit.add(in[i],1),bit.add(out[i],-1);
        for(auto p : vec) modify(p.first,p.second);
        scanf("%d",&q);
        printf("Case #%d:\n",kase-T+1);
        while(q--){
            int a,b,c;
            scanf("%d%d%d",&a,&b,&c);
            if(a==1) modify(b,c);
            else printf("%d\n",query(b,c));
        }
    }
    return 0;
}

树链修改和树链查询

牛客 A National Pandemic

题意：给定一棵树，接下来有三种操作。
第一种：指定一个点$x$和一个权值$w$，给所有点u的点权加上$w-dist(x,y)$。
第二种：将$x$的点权修改$min(0,val(x))$。
第三种：查询$x$的点权。

思路：第一种操作比较特殊，考虑能否将其转化成一般的树上修改。首先$w$是定值很好处理，只需要记录到目前为止的和即可。
$dist(x,y)$可以拆分成$dist(1,x)+dist(1,y)-2 \times dist(lca(x,y))$,前两个是定值都很好处理。考虑如何处理$dist(lca(x,y))$。我们可以对这棵树进行重链剖分，对于每个操作一，将$1-x$这条链上的点权加一。那么对于某个点$y，dist(lca(x,y))$就是重链剖分后$1-y$这条链上的权值和。操作二可以开一个数组记录差值即可。

代码：

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn = 5e4+10, mod = 1e9+7;
int _,n,m,u,v,dep[maxn],dfn[maxn],op,x,w;
int fa[maxn],siz[maxn],son[maxn],tim,top[maxn];
ll ans[maxn],sum[maxn<<2],tag[maxn<<2];
int head[maxn],to[maxn<<1],nex[maxn<<1],tot;

void add(int u, int v){
    to[++tot] = v,nex[tot] = head[u],head[u] = tot;
}

void dfs1(int u,int f){
    dep[u] = dep[f] + 1;
    fa[u] = f, siz[u] = 1;
    for (int i = head[u]; i; i = nex[i])if(to[i]^f){
        dfs1(to[i],u); siz[u] += siz[to[i]];
        if(siz[to[i]]>siz[son[u]]) son[u] = to[i];
    }
}

void dfs2(int u,int t){
    top[u] = t,dfn[u] = ++tim;
    if(!son[u]) return void();
    dfs2(son[u],t);
    for(int i = head[u]; i; i = nex[i])
        if(to[i]^fa[u]&&to[i]^son[u]) dfs2(to[i],to[i]);
}

void push_up(int rt) { sum[rt] = sum[rt<<1] + sum[rt<<1|1]; }

void build(int l,int r,int rt){
    tag[rt] = sum[rt] = 0;
    if (l == r) return void();
    int m = (l+r)>>1;
    build(l,m,rt<<1);
    build(m+1,r,rt<<1|1);
}

void push_down(int l,int r,int rt){
    if(!tag[rt]) return void();
    int m = (l+r)>>1;
    tag[rt<<1] += tag[rt];
    tag[rt<<1|1] += tag[rt];
    sum[rt<<1] += 1ll*(m-l+1)*tag[rt];
    sum[rt<<1|1] += 1ll*(r-m)*tag[rt];
    tag[rt] = 0;
}

void modify(int l,int r,int rt,int L,int R){
    if(l>R||r<L) return void();
    if(l>=L&&r<=R){
        tag[rt]++,sum[rt] += r-l+1;
        return void();
    }
    int m = (l+r)>>1;
    push_down(l,r,rt);
    modify(l,m,rt<<1,L,R);
    modify(m+1,r,rt<<1|1,L,R);
    push_up(rt);
}

ll query(int l,int r,int rt,int L,int R){
    if(l>R||r<L) return 0;
    if(l>=L&&r<=R) return sum[rt];
    ll ret = 0;
    int m = (l+r)>>1;
    push_down(l,r,rt);
    ret = query(l,m,rt<<1,L,R)+query(m+1,r,rt<<1|1,L,R);
    push_up(rt);
    return ret;
}

void change_chain(int x,int y){
    while (top[x]^top[y]){
        if(dep[top[x]]<dep[top[y]]) swap(x,y);
        modify(1,n,1,dfn[top[x]],dfn[x]);
        x = fa[top[x]];
    }
    if(dep[x]>dep[y]) swap(x,y);
    modify(1,n,1,dfn[x],dfn[y]);
}

ll query_chain(int x,int y){
    ll ret = 0;
    while (top[x]^top[y]){
        if(dep[top[x]]<dep[top[y]]) swap(x,y);
        ret += query(1,n,1,dfn[top[x]],dfn[x]);
        x = fa[top[x]];
    }
    if(dep[x]>dep[y]) swap(x,y);
    return ret+query(1,n,1,dfn[x],dfn[y]);
}

void init(int n){
    tim = tot = 0; build(1,n,1);
    for (int i = 1; i <= n; i++) 
        head[i] = ans[i] = son[i] = 0;
}

int main(){
    for (scanf("%d", &_);_;_--){
        scanf("%d%d",&n,&m); init(n);
        for (int i = 2; i <= n; i++){
            scanf("%d%d",&u,&v);
            add(u,v),add(v,u);
        }
        ll sum = 0,cnt = 0,tmp;
        dfs1(1,0),dfs2(1,1);
        for(int i = 1; i <= m; i++){
            scanf("%d", &op);
            if (op == 1){
                scanf("%d%d",&x,&w);
                sum += w-dep[x];
                change_chain(1,x);
                cnt++;
            }
            else if(op == 2){
                scanf("%d",&x);
                tmp = ans[x]-cnt*dep[x]+sum+2ll*query_chain(1,x);
                if(tmp > 0) ans[x] -= tmp;
            }
            else{
                scanf("%d", &x);
                printf("%lld\n",ans[x]-cnt*dep[x]+sum+2ll*query_chain(1,x));
            }
        }
    }
    return 0;
}

树上单点修改和半径求和

bzoj3730震波

题意： 给定一棵树，有两种操作。
第一种操作是求所有到点$x$的距离不超过$k$的点的权值和。
第二种操作是把点$x$的权值修改为$y$。

思路： 动态点分治/点分树板子题。

代码：

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn = 1e5+10;
int n,m,cnt,head[maxn],to[maxn<<1],nex[maxn<<1],id[maxn<<1];
int ans,u,v,op,x,y,val[maxn],del[maxn],siz[maxn];
void addedge(int u,int v,int w){
    to[++cnt] = v,id[cnt] = w,nex[cnt] = head[u],head[u] = cnt;
    to[++cnt] = u,id[cnt] = w,nex[cnt] = head[v],head[v] = cnt;
}
void get_size(int u,int fa){
    siz[u] = 1;
    for(int i = head[u]; i; i = nex[i]){
        if(to[i]==fa||del[to[i]]) continue;
        get_size(to[i],u),siz[u] += siz[to[i]];
    }
}
int get_gra(int u,int fa,int all){
    for(int i = head[u]; i; i = nex[i]){
        if(to[i]==fa||del[to[i]]) continue;
        if(siz[to[i]] > all/2) return get_gra(to[i],u,all);
    }
    return u;
}
struct node { int u,v,d; };
vector<node> path[maxn];
void get_path(int u,int fa,node p){
    path[u].push_back(p);
    for(int i = head[u]; i; i = nex[i]){
        if(to[i]==fa||del[to[i]]) continue;
        get_path(to[i],u,node{p.u,p.v,p.d+1});
    }
}
void div_tree(int u){
    get_size(u,0),u = get_gra(u,0,siz[u]),del[u] = true;
    path[u].push_back(node{u,0,0});
    for(int i = head[u]; i; i = nex[i]){
        if(del[to[i]]) continue;
        get_path(to[i],u,node{u,id[i],1});
    }
    for(int i = head[u]; i; i = nex[i])
        if(!del[to[i]]) div_tree(to[i]);
}
int root[2][maxn],ls[maxn<<7],rs[maxn<<7],sum[maxn<<7],tot;
int newnode() { tot++; ls[tot] = rs[tot] = sum[tot] = 0; return tot; }
void modify(int &p,int l,int r,int x,int y){
    if(!p) p = newnode();
    sum[p] += y;
    if(l==r) return void();
    int m = (l+r)>>1;
    if(x<=m) modify(ls[p],l,m,x,y);
    else modify(rs[p],m+1,r,x,y);
}
int query(int p,int l,int r,int L,int R){
    if(!p||l>R||r<L) return 0;
    if(l>=L&&r<=R) return sum[p];
    int m = (l+r)>>1;
    return query(ls[p],l,m,L,R)+query(rs[p],m+1,r,L,R);
}
void modify(int x,int y){
    for(auto p : path[x]){
        modify(root[0][p.u],0,n,p.d,y);
        if(p.v) modify(root[1][p.v],0,n,p.d,y);
    }
}
int query(int x,int y){
    int ret = 0;
    for(auto p : path[x]){
        ret += query(root[0][p.u],0,n,0,y-p.d);
        if(p.v) ret -= query(root[1][p.v],0,n,0,y-p.d);
    }
    return ret;
}
void init(){
    div_tree(1);
    for(int i = 1; i <= n; i++) modify(i,val[i]);
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i = 1; i <= n; i++) scanf("%d",val+i);
    for(int i = 2; i <= n; i++) scanf("%d%d",&u,&v),addedge(u,v,i);
    init();
    while(m--){
        scanf("%d%d%d",&op,&x,&y);
        x ^= ans,y ^= ans;
        if(op) modify(x,y-val[x]),val[x] = y;
        else printf("%d\n",ans = query(x,y));
    }
    return 0;
}
